```{r include = FALSE}
library(quantmod)
library(dygraphs)

stock1 <- getSymbols(params$symbol1, auto.assign = FALSE,
                    from = "2016-01-01", end = Sys.Date())
stock2 <- getSymbols(params$symbol2, auto.assign = FALSE,
                    from = "2016-01-01", end = Sys.Date())
```
